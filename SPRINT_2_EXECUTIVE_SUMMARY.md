  // Mobile layout
  return _buildMobileLayout();
}

// 3. Detect Orientation

  MediaQuery.of(context).orientation == Orientation.portrait;

  setState(() {
    // Trigger rebuild with new layout
  });
}

  children: [

    Expanded(child: content),  // Takes remaining space
